📋 Instrucciones: configurar Google Sheets (5 minutos)
1. Crea un Google Sheet nuevo
Ve a sheets.google.com → crea una hoja en blanco → ponle el nombre que quieras (ej. "Gestión Operaciones Riesgo")

2. Abre el editor de Apps Script
Extensiones → Apps Script → borra el código que viene por defecto

3. Pega este código: 
function doPost(e) {
  try {
    var params = JSON.parse(e.postData.contents);
    var result = handleRequest(params);
    return ContentService
      .createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({error: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function handleRequest(params) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var action = params.action;

  // ── Hoja de datos JSON (motor interno) ─────────────────────────────────────
  var sheet = ss.getSheetByName('GestionOps');
  if (!sheet) {
    sheet = ss.insertSheet('GestionOps');
    sheet.appendRow(['Clave', 'Datos', 'Actualizado']);
    sheet.setFrozenRows(1);
  }
  var data = sheet.getDataRange().getValues();

  if (action === 'set') {
    for (var i = 1; i < data.length; i++) {
      if (data[i][0] === params.key) {
        sheet.getRange(i+1,2).setValue(params.value);
        sheet.getRange(i+1,3).setValue(new Date().toISOString());
        return {success:true};
      }
    }
    sheet.appendRow([params.key, params.value, new Date().toISOString()]);
    return {success:true};
  }
  if (action === 'get') {
    for (var j = 1; j < data.length; j++) {
      if (data[j][0] === params.key) return {key:params.key, value:data[j][1]};
    }
    return {key:params.key, value:null};
  }
  if (action === 'list') {
    var prefix = params.prefix || '';
    var keys = [];
    for (var k = 1; k < data.length; k++) {
      if (data[k][0] && data[k][0].toString().startsWith(prefix)) keys.push(data[k][0].toString());
    }
    return {keys:keys};
  }
  if (action === 'ping') return {ok:true};

  // ── Hoja de registro histórico (legible, por fila) ─────────────────────────
  if (action === 'logRows') {
    var COLS = ['Fecha','Analista','Producto','Tipo ID','Identificador',
                'Estado','Urgencia','Observación','Hora Asignación',
                'Hora Completado','Confirmado por','ID Operación'];
    var log = ss.getSheetByName('RegistroHistorico');
    if (!log) {
      log = ss.insertSheet('RegistroHistorico');
      log.appendRow(COLS);
      log.setFrozenRows(1);
      var hdr = log.getRange(1,1,1,COLS.length);
      hdr.setBackground('#002960');
      hdr.setFontColor('#FFFFFF');
      hdr.setFontWeight('bold');
    }
    var logData = log.getDataRange().getValues();
    var uidCol = 11; // columna L (índice 0 = col A)
    var rows = params.rows;
    for (var r = 0; r < rows.length; r++) {
      var op = rows[r];
      var rowArr = [
        op.date        || '',
        op.analista    || '',
        op.productName || '',
        op.idType      || '',
        op.identificador || '(pendiente)',
        op.estado      || 'PENDIENTE',
        op.tier        || 'normal',
        op.obs_seguimiento || '',
        op.asignado_en  ? new Date(op.asignado_en).toLocaleString('es-CL')  : '',
        op.completado_en? new Date(op.completado_en).toLocaleString('es-CL'): '',
        op.confirmedBy || '',
        op.uid         || ''
      ];
      var found = false;
      for (var m = 1; m < logData.length; m++) {
        if (logData[m][uidCol] === op.uid) {
          log.getRange(m+1, 1, 1, COLS.length).setValues([rowArr]);
          logData[m] = rowArr;
          found = true;
          break;
        }
      }
      if (!found) {
        log.appendRow(rowArr);
        logData.push(rowArr);
      }
    }
    return {success:true, rows:rows.length};
  }

  return {error:'Acción no reconocida: ' + action};
}
4. Publica como aplicación web
Haz clic en Implementar → Nueva implementación → tipo: Aplicación web
Ejecutar como: Yo · Quién tiene acceso: Cualquier usuario → Implementar

5. Copia el URL del script
Copia el URL que aparece (empieza con https://script.google.com/macros/s/) y pégalo arriba.

6. Comparte la hoja con tu equipo
En Google Sheets → Compartir → agrega los emails de tu equipo con permiso de edición.
